<script>
// Update Foto
document.getElementById('upload').addEventListener('change', function(e) {
const reader = new FileReader();
reader.onload = function() {
document.getElementById('user-photo').src = reader.result;
}
reader.readAsDataURL(e.target.files[0]);
});

// Update Nama
document.getElementById('input-name').addEventListener('input', function(e) {
document.getElementById('name-preview').innerText = e.target.value || "Nama Anda";
});

// Ganti Tema
function setTheme(theme) {
const card = document.getElementById('product-canvas');
card.className = '';
if(theme !== 'sage') card.classList.add('theme-' + theme);
}

// Generate High Quality Image
function generate() {
const btn = document.querySelector('.btn-save');
btn.innerText = "Sedang Membuat...";
html2canvas(document.querySelector("#product-canvas"), {
scale: 3,
useCORS: true
}).then(canvas => {
const a = document.createElement('a');
a.href = canvas.toDataURL("image/png");
a.download = "Kartu_Lebaran_Pro.png";
a.click();
btn.innerText = "Download Gambar (.PNG)";
});
}
